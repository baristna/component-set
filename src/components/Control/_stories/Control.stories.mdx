# Control
Controls allow users to take actions, and make choices, with a single tap.

<br />

<Story name='Overview'>
  <Control type='checkbox' />
</Story>

<br /><br />

## Usage

```jsx
import { Control } from 'component-set';

<Control type='[TYPE]' />
```

<br /><br />

## Checkbox
<Story name='Checkbox'>
{() => {
  const [Checkbox1, setCheckbox1] = useState(false);
  const [Checkbox2, setCheckbox2] = useState(false);
  return (
    <Div mb={2}>
      <Div>
        <Control id='chk1' checked={Checkbox1} onChange={setCheckbox1} type='checkbox' />
      </Div>
      <Div>
        <Control id='chk2' checked={Checkbox2} onChange={setCheckbox2} type='checkbox' />
      </Div>
    </Div>
  )
}}
</Story>

```jsx
const [radio1, setRadio1] = useState('');

<Control id='rd1' value='true' checked={radio1} onChange={setRadio1} type='radio' />
<Control id='rd2' value='false' checked={radio1} onChange={setRadio1} type='radio' />
```
